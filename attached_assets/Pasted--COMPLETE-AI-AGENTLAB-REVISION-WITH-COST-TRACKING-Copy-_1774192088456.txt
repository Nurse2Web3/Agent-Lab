const express = require('express');
const cors = require('cors');

const Anthropic = require('@anthropic-ai/sdk');

const app = express();

// === PRICING TIERS ===
const PRICING = {
  trial: { limit: 10, models: ['gpt-4o-mini', 'claude-3-5-sonnet', 'grok-beta'] },
  pro: { limit: 100, models: ['gpt-4o-mini', 'claude-3-5-sonnet', 'grok-beta'] },
  premium: { limit: 500, models: ['gpt-4o-mini', 'claude-3-5-sonnet', 'grok-beta'] }
};

// === MODEL COST DATABASE ($/M tokens) ===
const MODEL_COSTS = {
  'gpt-4o-mini': { input: 0.15, output: 0.60 },
  'claude-3-5-sonnet-20241022': { input: 3.00, output: 15.00 },
  'grok-beta': { input: 5.00, output: 15.00 }
};

// === WORKFLOW TEMPLATES ===
const WORKFLOWS = {
  'support': {
    name: 'Customer Support Replies',
    template: `Customer: {input}
Write a professional, empathetic support reply in under 100 words. Match our brand voice: helpful, concise, solution-focused. End with next steps.`,
    metrics: ['quality', 'speed', 'cost']
  },
  'summarize': {
    name: 'Content Summarization',
    template: `Summarize this content in 3 bullet points max. Keep key facts, remove fluff: {input}`,
    metrics: ['quality', 'speed', 'cost']
  }
};

// Rate limiting per tier
const createLimiter = (limit) => rateLimit({
  windowMs: 24 * 60 * 60 * 1000,
  max: limit,
  message: `Tier limit (${limit}) reached. Upgrade at aiagentlab.com/pricing`
});

app.use('/api/trial', createLimiter(10));
app.use('/api/pro', createLimiter(100));
app.use('/api/premium', createLimiter(500));

// Init clients
function initClients(apiKeys) {
  const clients = {};
  if (apiKeys.openai) clients.openai = new OpenAI({ apiKey: apiKeys.openai });
  if (apiKeys.anthropic) clients.anthropic = new Anthropic({ apiKey: apiKeys.anthropic });
  if (apiKeys.xai) clients.grok = new OpenAI({ apiKey: apiKeys.xai, baseURL: 'https://api.x.ai/v1' });
  return clients;
}

// === MAIN COMPARE ENDPOINT ===
app.post('/api/:tier/compare', async (req, res) => {
  try {
    const { tier } = req.params;
    const { workflow, input, apiKeys } = req.body;
    
    if (!WORKFLOWS[workflow]) return res.status(400).json({ error: 'Use "support" or "summarize"' });
    if (!PRICING[tier]) return res.status(400).json({ error: 'Use /trial, /pro, or /premium' });
    
    const clients = initClients(apiKeys);
    if (Object.keys(clients).length === 0) return res.status(400).json({ error: 'Need 1+ API key' });
    
    const template = WORKFLOWS[workflow].template;
    const prompt = template.replace('{input}', input || 'Sample customer message');
    
    // Test all 3 models
    const results = await Promise.allSettled([
      testModel(clients.openai, 'gpt-4o-mini', prompt),
      testModel(clients.anthropic, 'claude-3-5-sonnet-20241022', prompt, 'Claude 3.5 Sonnet'),
      testModel(clients.grok, 'grok-beta', prompt, 'Grok Beta')
    ]);
    
    const comparison = results
      .map((result, i) => ({ model: ['GPT-4o-mini', 'Claude 3.5 Sonnet', 'Grok Beta'][i], ...result.value }))
      .filter(r => r.response);
    
    const winner = findWinner(comparison);
    
    res.json({
      tier,
      workflow: WORKFLOWS[workflow].name,
      prompt,
      comparison,
      winner,
      csvExport: generateCSV(comparison, winner)
    });
    
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

// === MODEL TESTING WITH COST ===
async function testModel(client, modelName, prompt, displayName = modelName) {
  if (!client) return null;
  
  const start = Date.now();
  let response, inputTokens = 100, outputTokens = 0;
  
  try {
    if (client.openai) {
      const completion = await client.chat.completions.create({
        model: modelName,
        messages: [{ role: 'user', content: prompt }],
        max_tokens: 150
      });
      response = completion.choices[0].message.content;
      inputTokens = completion.usage.prompt_tokens;
      outputTokens = completion.usage.completion_tokens;
    } else if (client.anthropic) {
      const completion = await client.messages.create({
        model: modelName,
        max_tokens: 150,
        messages: [{ role: 'user', content: prompt }]
      });
      response = completion.content[0].text;
      inputTokens = completion.usage.input_tokens;
      outputTokens = completion.usage.output_tokens;
    }
  } catch (e) {
    return { error: e.message };
  }
  
  const costData = MODEL_COSTS[modelName];
  const dollarCost = ((inputTokens * costData.input) + (outputTokens * costData.output)) / 1000000;
  
  return {
    model: displayName,
    response: response?.substring(0, 200) + '...',
    speed: Date.now() - start + 'ms',
    tokens: inputTokens + outputTokens,
    dollarCost: `$${dollarCost.toFixed(4)}`,
    qualityScore: simpleQualityScore(response),
    costPerQuality: (dollarCost / simpleQualityScore(response)).toFixed(4)
  };
}

function simpleQualityScore(response) {
  const words = response.split(' ').length;
  const hasBullets = response.includes('•') || response.includes('-') || response.includes('*');
  return Math.min((words / 20) + (hasBullets ? 2 : 0), 10);
}

function findWinner(results) {
  return results.reduce((winner, current) => {
    const winnerScore = current.qualityScore - (parseFloat(current.dollarCost.slice(1)) * 10);
    const currentScore = current.qualityScore - (parseFloat(current.dollarCost.slice(1)) * 10);
    return currentScore > winnerScore ? current : winner;
  });
}

function generateCSV(results, winner) {
  return 'Model,Speed,Total Tokens,Dollar Cost,Quality,Cost/Quality\n' + 
    results.map(r => `${r.model},"${r.speed}",${r.tokens},"${r.dollarCost}",${r.qualityScore},${r.costPerQuality}`).join('\n');
}
